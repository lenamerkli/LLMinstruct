from bs4 import BeautifulSoup, Comment
import re
from pprint import pprint


def reduce_html(html: str) -> str:
    soup = BeautifulSoup(html, 'html.parser')
    for tag in soup(['style', 'script', 'head', 'meta', 'link', 'noscript']):
        tag.decompose()
    for element in soup.find_all(True):
        for attr in ['style', 'face', 'border', 'align', 'nowrap', 'size', 'color', 'width', 'bgcolor', 'fgcolor', 'cellpadding', 'cellspacing', 'rules']:
            if attr in element.attrs:
                del element.attrs[attr]
    for comment in soup.find_all(string=lambda text: isinstance(text, Comment)):
        comment.extract()
    cleaned = soup.prettify()
    return cleaned


def parse(html: str) -> dict:
    parser = TimetableParser(html)
    return parser.parse()


class TimetableParser:
    """
    Parses a school timetable from a specific HTML format into a structured format
    suitable for database insertion.
    """

    def __init__(self, html_content: str):
        """
        Initializes the parser with the HTML content.

        Args:
            html_content: A string containing the HTML of the timetable page.
        """
        self.soup = BeautifulSoup(html_content, 'html.parser')
        self.base_plan = []
        self.notes = []

    def parse(self) -> dict:
        """
        Executes the parsing of the entire HTML document.

        Returns:
            A dictionary containing the 'base_plan' and 'notes' data.
        """
        class_name = self._parse_header()
        self._parse_main_table(class_name)
        self._parse_notes_table()

        return {
            "base_plan": self.base_plan,
            "notes": self.notes
        }

    def _parse_header(self) -> str:
        """Parses the header information (class name, etc.)."""
        center_tag = self.soup.find('center')
        # The class name is the second bold tag within font tags in center
        font_tags = center_tag.find_all('font')
        class_name = None
        for font in font_tags:
            b_tag = font.find('b')
            if b_tag and b_tag.text.strip():
                class_name = b_tag.text.strip()
                break
        return class_name

    def _parse_main_table(self, class_name: str):
        """Parses the main grid timetable."""
        main_table = self.soup.select_one('body > center > table')
        if not main_table:
            return

        rows = main_table.find_all('tr', recursive=False)

        # Extract days from original header before grid expansion
        header_tr = rows[0]
        days = [td.text.strip() for td in header_tr.find_all('td') if td.get('colspan') == '12']

        # Build a grid map to handle complex rowspans and colspans
        grid_map = []
        for r, tr in enumerate(rows):
            while len(grid_map) <= r:
                grid_map.append([None] * 70)  # Pre-allocate enough columns

            for td in tr.find_all('td', recursive=False):
                c = 0
                while grid_map[r][c] is not None:
                    c += 1

                rowspan = int(td.get('rowspan', 1))
                colspan = int(td.get('colspan', 1))

                for i in range(rowspan):
                    for j in range(colspan):
                        if len(grid_map) <= r + i:
                            grid_map.append([None] * 70)
                        grid_map[r + i][c + j] = td

        processed_cells = set()  # Use a set to avoid processing a cell multiple times

        for r in range(1, len(grid_map)):
            for c in range(1, len(grid_map[r])):
                lesson_cell = grid_map[r][c]

                if not lesson_cell or id(lesson_cell) in processed_cells:
                    continue

                processed_cells.add(id(lesson_cell))

                # This is the top-left corner of a new lesson/event.
                # Determine its day based on its column index.
                day_index = (c - 1) // 12
                if day_index >= len(days): continue
                day = days[day_index]

                # Determine start time by finding the time-defining cell for this row.
                r_start = r
                while r_start >= 1:
                    time_def_cell = grid_map[r_start][0]
                    if time_def_cell and time_def_cell.get('rowspan') == '2':
                        break
                    r_start -= 1

                start_time_text = grid_map[r_start][0].text.strip()
                start_time, _ = start_time_text.split('-')

                # Determine end time based on the lesson's rowspan.
                rowspan = int(lesson_cell.get('rowspan', 2))

                # Extract lesson details from the cell's <font> tags.
                texts = [f.text.strip() for f in lesson_cell.find_all('font') if f.text.strip()]
                if not texts: continue  # It's an empty cell

                subject = texts[0] if len(texts) > 0 else None
                teacher = texts[1] if len(texts) > 1 else None
                room = None
                note_id = None

                if len(texts) == 3:
                    room = texts[2]
                elif len(texts) == 4:
                    room = texts[2]
                    # Note ID is usually numeric and enclosed in parentheses
                    if re.match(r'\d+\)', texts[3]):
                        note_id = texts[3].replace(')', '')

                # If lesson spans multiple time slots, duplicate it for each slot
                num_slots = rowspan // 2
                for slot_idx in range(num_slots):
                    slot_row = r_start + slot_idx * 2
                    slot_time_text = grid_map[slot_row][0].text.strip()
                    slot_start_time, slot_end_time = slot_time_text.split('-')

                    self.base_plan.append({
                        'class_name': class_name,
                        'day': day,
                        'start_time': slot_start_time,
                        'end_time': slot_end_time,
                        'subject': subject,
                        'teacher': teacher,
                        'room': room,
                        'note_id': note_id,
                    })

    def _parse_notes_table(self):
        """Parses the table containing supplementary notes."""
        # The notes table is the second <table> in the <center> tag.
        tables = self.soup.select('body > center > table')
        if len(tables) < 2:
            return

        notes_table = tables[1]
        rows = notes_table.find_all('tr')[1:]  # Skip header row

        # Track the last seen note_id for forward-filling
        last_note_id = None

        for row in rows:
            cells = [td.text.strip() for td in row.find_all('td')]
            if len(cells) < 20:
                continue

            # Process the left-side entry in the row
            if cells[0]:
                # This row has its own note_id
                current_note_id = cells[0].replace(')', '')
                last_note_id = current_note_id
            elif cells[1] and last_note_id:
                # This row has no note_id but has details, use the last seen note_id
                current_note_id = last_note_id
            else:
                current_note_id = None

            if cells[1]:  # Process if there's details data
                self.notes.append({
                    'note_id': current_note_id,
                    'details': cells[1],  # Le.,Fa.,Rm.
                    'classes': cells[2],
                    'time': cells[3],
                    'week': cells[4],  # Schulwoche
                    'studt': cells[5],
                    'text1': cells[6],  # Text
                    'band': cells[7],
                    'text2': cells[8],  # ZeilenText-2
                    'group': cells[9],  # Schülergruppe
                    'base_class': self._parse_header()  # Add base class
                })

            # Process the right-side entry in the row
            if cells[10]:
                # This row has its own note_id for the right side
                current_note_id_right = cells[10].replace(')', '')
                last_note_id_right = current_note_id_right
            elif cells[11] and last_note_id_right:
                # This row has no note_id but has details, use the last seen note_id
                current_note_id_right = last_note_id_right
            else:
                current_note_id_right = None

            if cells[11]:  # Process if there's details data
                self.notes.append({
                    'note_id': current_note_id_right,
                    'details': cells[11],
                    'classes': cells[12],
                    'time': cells[13],
                    'week': cells[14],
                    'studt': cells[15],
                    'text1': cells[16],
                    'band': cells[17],
                    'text2': cells[18],
                    'group': cells[19],
                    'base_class': self._parse_header()  # Add base class
                })
