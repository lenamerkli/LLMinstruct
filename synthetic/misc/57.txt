Here's a small Python script that attempts to **guess a person's name from their email address** using common email patterns. It uses simple heuristics like splitting the email and looking for name-like parts.

```python
def guess_name_from_email(email):
    # Normalize the email
    email = email.strip().lower()

    # Split by @ and take the local part
    local_part = email.split('@')[0]

    # Remove dots from local part (e.g., john.doe -> john doe)
    local_part = local_part.replace('.', ' ')

    # Split into possible name components
    parts = local_part.split()

    # Filter out numbers and obvious non-name parts
    filtered_parts = [part for part in parts if not part.isdigit()]

    # Try to extract first and last name
    if len(filtered_parts) == 1:
        # Single part: try to guess it's the first name
        first_name = filtered_parts[0].title()
        return f"{first_name} Alias"

    elif len(filtered_parts) >= 2:
        # Try to extract first and last name
        first_name = filtered_parts[0].title()
        last_name = filtered_parts[-1].title()

        # If last name is just a number or not plausible, fallback
        if last_name.isdigit() or len(last_name) < 2:
            return f"{first_name} Alias"

        # If first name is just a number or too short
        if first_name.isdigit() or len(first_name) < 2:
            return f"{last_name} Alias"

        return f"{first_name} {last_name}"

    else:
        return "Alias"

# Example usage
if __name__ == "__main__":
    email = input("Enter an email address: ")
    print(guess_name_from_email(email))
```

### ðŸ§  How it works:

- **Normalize**: Convert email to lowercase and strip whitespace.
- **Extract local part**: Everything before `@`.
- **Remove dots**: Treat `john.doe` as `john doe`.
- **Split and filter**: Remove digits and keep plausible name parts.
- **Guess name**:
  - If one part â†’ assume it's a first name â†’ `Firstname Alias`.
  - If multiple parts â†’ use first as first name, last as last name â†’ `Firstname Lastname`.
  - Fallback â†’ `"Alias"`.

import re

def guess_name_from_email(email):
    # Extract the part before the @
    username = email.split('@')[0]

    # Remove trailing numbers (e.g., Joe43_Doe â†’ Joe_Doe)
    username = re.sub(r'\d+$', '', username)

    # Try splitting by common separators
    for sep in ['.', '_', '-']:
        if sep in username:
            parts = username.split(sep)
            # Only use first two parts as first and last name
            if len(parts) >= 2:
                first, last = parts[0], parts[1]
                return f"{first.capitalize()} {last.capitalize()}"

    # If no separator found, return Alias
    return "Alias"

# Example usage:
