```sql
-- Notes table (for 'notes' table)
CREATE TABLE notes (
    note_id INT PRIMARY KEY,
    base_class VARCHAR(20),
    classes VARCHAR(100), -- Can hold comma-separated lists like 'U25a, U25b'
    text1 VARCHAR(100),
    text2 VARCHAR(100),
    details VARCHAR(100),
    week VARCHAR(20), -- e.g., '1-46'
    band VARCHAR(50),
    group_name VARCHAR(50), -- Renamed from 'group' to avoid reserved keyword issues
    studt VARCHAR(50),
    time VARCHAR(20) -- Using VARCHAR due to empty values in sample; could be TIME if validated
);

-- Schedule/timetable entries table
CREATE TABLE schedule (
    class_name VARCHAR(20) NOT NULL,
    day VARCHAR(10) NOT NULL, -- e.g., 'Mo', could also be full day names
    start_time TIME NOT NULL,
    end_time TIME NOT NULL,
    subject VARCHAR(20) NOT NULL,
    teacher VARCHAR(50),
    room VARCHAR(20),
    note_id INT, -- Foreign key to notes; NULL if no note is associated
    PRIMARY KEY (class_name, day, start_time, subject), -- Composite key assumes this combo is unique per slot
    CONSTRAINT fk_schedule_note FOREIGN KEY (note_id) REFERENCES notes(note_id)
);
```
